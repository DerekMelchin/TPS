460:     /**

462:      * There is also a check if we use the special tilde ~

464:      * @param string $text
465:      * @param string $currentMode
466:      * @return string

469:         if ($this->tilde) {

472:             while (($pos = strpos($text, '~', $previousPos)) !== false) {
473:                 $tildeData = $this->extractTilde($text, $pos);

475:                 $simpleTilde = ($tildeData === '~~');
476:                 if ($simpleTilde && $currentMode !== 'B') {
477:                     throw new BCGParseException('code128', 'The Table ' . $currentMode . ' doesn\'t contain the character ~.');
478:                 }
479: 
480:                 // At this point, we know we have ~Fx
481:                 if ($tildeData !== '~F1' && $currentMode === 'C') {
482:                     // The mode C doesn't support ~F2, ~F3, ~F4
483:                     throw new BCGParseException('code128', 'The Table C doesn\'t contain the function ' . $tildeData . '.');
484:                 }

494:                 $sequence .= '.';
495:                 $sequence .= (!$simpleTilde) ? 'F' : '';

497:             }

499:             // Flushing
500:             $length = strlen($text) - $previousPos;
501:             if ($currentMode === 'C') {
502:                 if ($length % 2 === 1) {
503:                     throw new BCGParseException('code128', 'The text "'.$text.'" must have an even number of character to be encoded in Table C.');
504:                 }
505:             }

507:             $sequence .= str_repeat('.', $length);
