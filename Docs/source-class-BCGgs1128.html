331:     /**
332:      * Splits the id and the content for each application identifiers (AIs).
333:      *
334:      * @param string $text
335:      * @param int $cmpt
336:      * @return bool
337:      */
338:     private function parseContent($text) {
339:         /* $yAlreadySet has 3 states: */
340:         /* null: There is no variable in the ID; true: the variable is already set; false: the variable is not set yet; */
341:         $content = NULL;
342:         $yAlreadySet = NULL;
343:         $realNameId = NULL;
344:         $separatorsFound = 0;
345:         $checksumAdded = 0;
346:         $decimalPointRemoved = 0;
347:         $toParse = str_replace('~F1', chr(29), $text);
348:         $nbCharToParse = strlen($toParse);
349:         $nbCharId = 0;
350:         $isFormated = $toParse[0] === '(' ? true : false;
351:         $maxCharId = $isFormated ? self::MAX_ID_FORMATED : self::MAX_ID_NOT_FORMATED;

355:         if ($id === false) {
356:             if ($this->allowsUnknownIdentifier === false) {
357:                 return false;
358:             }
359: 
360:             $id = NULL;
361:             $nbCharId = 0;
362:             $content = $toParse;
363:         } else {
364:             $nbCharId = strlen($id) + ($isFormated ? 2 : 0);
365:             $n = min($this->identifiersAi[$realNameId][self::MAXLENGTH], $nbCharToParse);

367:         }
368:         
369:         if ($id !== NULL) {
370:             /* If we have an AI with an "y" var, we check if there is a decimal point in the next *MAXLENGTH* characters */
371:             /* if there is one, we take an extra character */
372:             if ($yAlreadySet !== NULL) {
373:                 if (strpos($content, '.') !== false || strpos($content, ',') !== false) {
374:                     $n++;
375:                     if ($n <= $nbCharToParse) {
376:                         /* We take an extra char */
377:                         $content = substr($toParse, $nbCharId, $n);
378:                     }
379:                 }
380:             }
381:         }
382:         
383:         /* We check for separator */
384:         $separator = strpos($content, chr(29));
385:         if ($separator !== false) {
386:             $content = substr($content, 0, $separator);
387:             $separatorsFound++;
388:         }
389:         
390:         if ($id !== NULL) {
391:             /* We check the conformity */
392:             if (!$this->checkConformity($content, $id, $realNameId)) {
393:                 return false;
394:             }
395: 
396:             /* We check the checksum */
397:             if (!$this->checkChecksum($content, $id, $realNameId, $checksumAdded)) {
398:                 return false;
399:             }
400: 
401:             /* We check the vars */
402:             if (!$this->checkVars($content, $id, $yAlreadySet, $decimalPointRemoved)) {
403:                 return false;
404:             }
405:         }
