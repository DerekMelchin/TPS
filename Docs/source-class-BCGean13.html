193:     /**
194:      * Overloaded method to calculate checksum.
195:      */
196:     protected function calculateChecksum() {
197:         // Calculating Checksum
198:         // Consider the right-most digit of the message to be in an "odd" position,
199:         // and assign odd/even to each character moving from right to left
200:         // Odd Position = 3, Even Position = 1
201:         // Multiply it by the number
202:         // Add all of that and do 10-(?mod10)
203:         $odd = true;
204:         $this->checksumValue = 0;
205:         $c = strlen($this->text);
206:         for ($i = $c; $i > 0; $i--) {
207:             if ($odd === true) {
208:                 $multiplier = 3;
209:                 $odd = false;
210:             } else {
211:                 $multiplier = 1;
212:                 $odd = true;
213:             }
214: 
215:             if (!isset($this->keys[$this->text[$i - 1]])) {
216:                 return;
217:             }
218: 
219:             $this->checksumValue += $this->keys[$this->text[$i - 1]] * $multiplier;
220:         }
221: 
222:         $this->checksumValue = (10 - $this->checksumValue % 10) % 10;
223:     }
