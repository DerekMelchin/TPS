122:         // The "C" checksum character is the modulo 11 remainder of the sum of the weighted
123:         // value of the data characters. The weighting value starts at "1" for the right-most
124:         // data character, 2 for the second to last, 3 for the third-to-last, and so on up to 20.

134:             $c = strlen($temp_text);
135: 
136:             // We don't display the K CheckSum if the original text had a length less than 10
137:             if ($c <= 10 && $z === 1) {
138:                 break;
139:             }

142:             for ($i = $c, $j = 0; $i > 0; $i--, $j++) {

144:                 if ($multiplier === 0) {

146:                 }

149:             }
